def merge_sort(arr):
    if len(arr) > 1:
        mid = len(arr) // 2
        left_half = arr[:mid]
        right_half = arr[mid:]
        merge_sort(left_half)
        merge_sort(right_half)
        i = j = k = 0
        while i < len(left_half) and j < len(right_half):
            if left_half[i] < right_half[j]:
                arr[k] = left_half[i]
                i += 1
            else:
                arr[k] = right_half[j]
                j += 1
            k += 1
        while i < len(left_half):
            arr[k] = left_half[i]
            i += 1
            k += 1
        while j < len(right_half):
            arr[k] = right_half[j]
            j += 1
            k += 1

def quick_sort(arr, low, high):
    if low < high:
        pivot = arr[high]
        i = low - 1
        for j in range(low, high):
            if arr[j] <= pivot:
                i = i + 1
                arr[i], arr[j] = arr[j], arr[i]
        arr[i + 1], arr[high] = arr[high], arr[i + 1]
        pi = i + 1
        quick_sort(arr, low, pi - 1)
        quick_sort(arr, pi + 1, high)

def heap_sort(arr):
    n = len(arr)
    for i in range(n // 2 - 1, -1, -1):
        largest = i
        l = 2 * i + 1
        r = 2 * i + 2
        if l < n and arr[i] < arr[l]:
            largest = l
        if r < n and arr[largest] < arr[r]:
            largest = r
        if largest != i:
            arr[i], arr[largest] = arr[largest], arr[i]
    for i in range(n - 1, 0, -1):
        arr[i], arr[0] = arr[0], arr[i]
        largest = 0
        l = 1
        r = 2
        if l < i and arr[0] < arr[l]:
            largest = l
        if r < i and arr[largest] < arr[r]:
            largest = r
        if largest != 0:
            arr[0], arr[largest] = arr[largest], arr[0]

def shell_sort(arr):
    n = len(arr)
    gap = n // 2
    while gap > 0:
        for i in range(gap, n):
            temp = arr[i]
            j = i
            while j >= gap and arr[j - gap] > temp:
                arr[j] = arr[j - gap]
                j -= gap
            arr[j] = temp
        gap //= 2
    return arr

def insertion_sort_advanced(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and key < arr[j]:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

def binary_search_recursive(arr, low, high, x):
    if high >= low:
        mid = (high + low) // 2
        if arr[mid] == x:
            return mid
        elif arr[mid] > x:
            return binary_search_recursive(arr, low, mid - 1, x)
        else:
            return binary_search_recursive(arr, mid + 1, high, x)
    else:
        return -1

def exponential_search(arr, n, x):
    if arr[0] == x:
        return 0
    i = 1
    while i < n and arr[i] <= x:
        i = i * 2
    low = i // 2
    high = min(i, n - 1)
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == x:
            return mid
        elif arr[mid] < x:
            low = mid + 1
        else:
            high = mid - 1
    return -1

def jump_search(arr, x, n):
    import math
    step = math.sqrt(n)
    prev = 0
    while arr[int(min(step, n) - 1)] < x:
        prev = step
        step += math.sqrt(n)
        if prev >= n:
            return -1
    while arr[int(prev)] < x:
        prev += 1
        if prev == min(step, n):
            return -1
    if arr[int(prev)] == x:
        return int(prev)
    return -1

def matrix_multiplication(A, B):
    result = [[0 for _ in range(len(B[0]))] for _ in range(len(A))]
    for i in range(len(A)):
        for j in range(len(B[0])):
            for k in range(len(B)):
                result[i][j] += A[i][k] * B[k][j]
    return result

def rotate_matrix_90_degrees(matrix):
    n = len(matrix)
    for i in range(n // 2):
        for j in range(i, n - i - 1):
            temp = matrix[i][j]
            matrix[i][j] = matrix[n - 1 - j][i]
            matrix[n - 1 - j][i] = matrix[n - 1 - i][n - 1 - j]
            matrix[n - 1 - i][n - 1 - j] = matrix[j][n - 1 - i]
            matrix[j][n - 1 - i] = temp
    return matrix

def generate_primes_sieve(limit):
    sieve = [True] * (limit + 1)
    p = 2
    while p * p <= limit:
        if sieve[p]:
            for i in range(p * p, limit + 1, p):
                sieve[i] = False
        p += 1
    primes = []
    for p in range(2, limit + 1):
        if sieve[p]:
            primes.append(p)
    return primes

def gcd_extended(a, b):
    if a == 0:
        return b, 0, 1
    gcd, x1, y1 = gcd_extended(b % a, a)
    x = y1 - (b // a) * x1
    y = x1
    return gcd, x, y

def fibonacci_memoization(n, memo={}):
    if n in memo:
        return memo[n]
    if n <= 0:
        return 0
    elif n == 1:
        return 1
    else:
        result = fibonacci_memoization(n - 1, memo) + fibonacci_memoization(n - 2, memo)
        memo[n] = result
        return result

def pascal_triangle(num_rows):
    triangle = []
    for i in range(num_rows):
        row = [None for _ in range(i + 1)]
        row[0], row[-1] = 1, 1
        for j in range(1, len(row) - 1):
            row[j] = triangle[i - 1][j - 1] + triangle[i - 1][j]
        triangle.append(row)
    return triangle

def base_converter(number, base):
    if number == 0:
        return "0"
    digits = []
    while number > 0:
        remainder = number % base
        if remainder >= 10:
            digits.append(chr(ord('A') + remainder - 10))
        else:
            digits.append(str(remainder))
        number //= base
    digits.reverse()
    return "".join(digits)

def is_palindrome_string(s):
    cleaned = ""
    for char in s:
        if char.isalnum():
            cleaned += char.lower()
    left = 0
    right = len(cleaned) - 1
    while left < right:
        if cleaned[left] != cleaned[right]:
            return False
        left += 1
        right -= 1
    return True

def group_anagrams(words):
    anagrams = {}
    for word in words:
        sorted_word = "".join(sorted(word))
        if sorted_word not in anagrams:
            anagrams[sorted_word] = []
        anagrams[sorted_word].append(word)
    result = []
    for key in anagrams:
        result.append(anagrams[key])
    return result

def compress_string(s):
    if not s:
        return ""
    compressed = []
    count = 1
    for i in range(1, len(s)):
        if s[i] == s[i - 1]:
            count += 1
        else:
            compressed.append(s[i - 1] + str(count))
            count = 1
    compressed.append(s[-1] + str(count))
    result = "".join(compressed)
    if len(result) < len(s):
        return result
    return s

def caesar_cipher(text, shift):
    result = ""
    for i in range(len(text)):
        char = text[i]
        if char.isupper():
            result += chr((ord(char) + shift - 65) % 26 + 65)
        elif char.islower():
            result += chr((ord(char) + shift - 97) % 26 + 97)
        else:
            result += char
    return result

def longest_common_prefix(strs):
    if not strs:
        return ""
    prefix = strs[0]
    for i in range(1, len(strs)):
        while strs[i].find(prefix) != 0:
            prefix = prefix[:-1]
            if not prefix:
                return ""
    return prefix

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def append(self, data):
        new_node = Node(data)
        if not self.head:
            self.head = new_node
            return
        last = self.head
        while last.next:
            last = last.next
        last.next = new_node

    def display(self):
        current = self.head
        elements = []
        while current:
            elements.append(current.data)
            current = current.next
        return elements

def reverse_linked_list(head):
    prev = None
    current = head
    while current is not None:
        next_node = current.next
        current.next = prev
        prev = current
        current = next_node
    return prev

def detect_cycle_linked_list(head):
    slow = head
    fast = head
    while fast is not None and fast.next is not None:
        slow = slow.next
        fast = fast.next.next
        if slow == fast:
            return True
    return False

def merge_sorted_linked_lists(l1, l2):
    dummy = Node(0)
    tail = dummy
    while l1 is not None and l2 is not None:
        if l1.data < l2.data:
            tail.next = l1
            l1 = l1.next
        else:
            tail.next = l2
            l2 = l2.next
        tail = tail.next
    if l1 is not None:
        tail.next = l1
    elif l2 is not None:
        tail.next = l2
    return dummy.next

class Stack:
    def __init__(self):
        self.items = []

    def is_empty(self):
        return len(self.items) == 0

    def push(self, item):
        self.items.append(item)

    def pop(self):
        if not self.is_empty():
            return self.items.pop()
        return None

    def peek(self):
        if not self.is_empty():
            return self.items[-1]
        return None

class Queue:
    def __init__(self):
        self.items = []

    def is_empty(self):
        return len(self.items) == 0

    def enqueue(self, item):
        self.items.insert(0, item)

    def dequeue(self):
        if not self.is_empty():
            return self.items.pop()
        return None

    def size(self):
        return len(self.items)

def evaluate_postfix(expression):
    stack = []
    for char in expression:
        if char.isdigit():
            stack.append(int(char))
        else:
            val1 = stack.pop()
            val2 = stack.pop()
            if char == '+':
                stack.append(val2 + val1)
            elif char == '-':
                stack.append(val2 - val1)
            elif char == '*':
                stack.append(val2 * val1)
            elif char == '/':
                stack.append(val2 / val1)
    return stack.pop()

def check_balanced_parentheses(expression):
    stack = []
    mapping = {')': '(', '}': '{', ']': '['}
    for char in expression:
        if char in mapping.values():
            stack.append(char)
        elif char in mapping.keys():
            if not stack or mapping[char] != stack.pop():
                return False
    return len(stack) == 0

class TreeNode:
    def __init__(self, key):
        self.left = None
        self.right = None
        self.val = key

def insert_bst(root, key):
    if root is None:
        return TreeNode(key)
    else:
        if root.val == key:
            return root
        elif root.val < key:
            root.right = insert_bst(root.right, key)
        else:
            root.left = insert_bst(root.left, key)
    return root

def inorder_traversal(root):
    result = []
    if root:
        result = inorder_traversal(root.left)
        result.append(root.val)
        result = result + inorder_traversal(root.right)
    return result

def max_depth_tree(root):
    if root is None:
        return 0
    else:
        left_depth = max_depth_tree(root.left)
        right_depth = max_depth_tree(root.right)
        if left_depth > right_depth:
            return left_depth + 1
        else:
            return right_depth + 1

def is_symmetric_tree(root):
    def is_mirror(t1, t2):
        if t1 is None and t2 is None:
            return True
        if t1 is None or t2 is None:
            return False
        return (t1.val == t2.val) and is_mirror(t1.right, t2.left) and is_mirror(t1.left, t2.right)
    if root is None:
        return True
    return is_mirror(root.left, root.right)

class GraphDFS:
    def __init__(self):
        self.graph = {}

    def add_edge(self, u, v):
        if u not in self.graph:
            self.graph[u] = []
        self.graph[u].append(v)

    def dfs_util(self, v, visited):
        visited.add(v)
        if v in self.graph:
            for neighbour in self.graph[v]:
                if neighbour not in visited:
                    self.dfs_util(neighbour, visited)

    def dfs(self, v):
        visited = set()
        self.dfs_util(v, visited)
        return visited

class GraphBFS:
    def __init__(self):
        self.graph = {}

    def add_edge(self, u, v):
        if u not in self.graph:
            self.graph[u] = []
        self.graph[u].append(v)

    def bfs(self, start):
        visited = set()
        queue = [start]
        visited.add(start)
        traversal = []
        while queue:
            s = queue.pop(0)
            traversal.append(s)
            if s in self.graph:
                for i in self.graph[s]:
                    if i not in visited:
                        queue.append(i)
                        visited.add(i)
        return traversal

def count_islands(grid):
    if not grid:
        return 0
    count = 0
    for i in range(len(grid)):
        for j in range(len(grid[0])):
            if grid[i][j] == '1':
                dfs_island(grid, i, j)
                count += 1
    return count

def dfs_island(grid, i, j):
    if i < 0 or j < 0 or i >= len(grid) or j >= len(grid[0]) or grid[i][j] != '1':
        return
    grid[i][j] = '0'
    dfs_island(grid, i + 1, j)
    dfs_island(grid, i - 1, j)
    dfs_island(grid, i, j + 1)
    dfs_island(grid, i, j - 1)

def dijkstra_algorithm(graph, start):
    distances = {node: float('infinity') for node in graph}
    distances[start] = 0
    visited = set()
    while len(visited) < len(graph):
        min_node = None
        for node in graph:
            if node not in visited:
                if min_node is None or distances[node] < distances[min_node]:
                    min_node = node
        if distances[min_node] == float('infinity'):
            break
        for neighbor, weight in graph[min_node].items():
            new_distance = distances[min_node] + weight
            if new_distance < distances[neighbor]:
                distances[neighbor] = new_distance
        visited.add(min_node)
    return distances

def longest_increasing_subsequence(arr):
    n = len(arr)
    dp = [1] * n
    for i in range(1, n):
        for j in range(0, i):
            if arr[i] > arr[j] and dp[i] < dp[j] + 1:
                dp[i] = dp[j] + 1
    maximum = 0
    for i in range(n):
        if maximum < dp[i]:
            maximum = dp[i]
    return maximum

def knapsack_problem(W, wt, val, n):
    K = [[0 for _ in range(W + 1)] for _ in range(n + 1)]
    for i in range(n + 1):
        for w in range(W + 1):
            if i == 0 or w == 0:
                K[i][w] = 0
            elif wt[i - 1] <= w:
                K[i][w] = max(val[i - 1] + K[i - 1][w - wt[i - 1]], K[i - 1][w])
            else:
                K[i][w] = K[i - 1][w]
    return K[n][W]

def coin_change(coins, amount):
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0
    for coin in coins:
        for x in range(coin, amount + 1):
            dp[x] = min(dp[x], dp[x - coin] + 1)
    if dp[amount] != float('inf'):
        return dp[amount]
    else:
        return -1

def edit_distance(word1, word2):
    m = len(word1)
    n = len(word2)
    dp = [[0 for _ in range(n + 1)] for _ in range(m + 1)]
    for i in range(m + 1):
        for j in range(n + 1):
            if i == 0:
                dp[i][j] = j
            elif j == 0:
                dp[i][j] = i
            elif word1[i - 1] == word2[j - 1]:
                dp[i][j] = dp[i - 1][j - 1]
            else:
                dp[i][j] = 1 + min(dp[i][j - 1], dp[i - 1][j], dp[i - 1][j - 1])
    return dp[m][n]

class BankAccount:

        self.owner = owner
        self.balance = balance

            self.balance += amount
            return True

class ShoppingCart:
    def __init__(self):
        self.items = {}

    def add_item(self, item_name, price, quantity=1):
        if item_name in self.items:
            self.items[item_name]['quantity'] += quantity
        else:
            self.items[item_name] = {'price': price, 'quantity': quantity}

    def remove_item(self, item_name):
        if item_name in self.items:
            del self.items[item_name]

        return total

def calculate_statistics(data):
    if not data:
        return None, None, None
    n = len(data)
    mean = sum(data) / n
    sorted_data = sorted(data)
    if n % 2 == 0:
        median = (sorted_data[n//2 - 1] + sorted_data[n//2]) / 2
    else:
        median = sorted_data[n//2]
    freq = {}
    for item in data:
        freq[item] = freq.get(item, 0) + 1
    max_freq = max(freq.values())
    mode = [k for k, v in freq.items() if v == max_freq]
    return mean, median, mode

def calculate_variance_and_std(data, sample=True):
    if len(data) < 2:
        return 0.0, 0.0
    n = len(data)
    mean = sum(data) / n
    deviations = [(x - mean) ** 2 for x in data]
    variance = sum(deviations) / (n - 1 if sample else n)
    std_dev = variance ** 0.5
    return variance, std_dev

def simple_linear_regression(x_values, y_values):
    n = len(x_values)
    sum_x = sum(x_values)
    sum_y = sum(y_values)
    sum_xy = sum(x * y for x, y in zip(x_values, y_values))
    sum_xx = sum(x ** 2 for x in x_values)
    slope = (n * sum_xy - sum_x * sum_y) / (n * sum_xx - sum_x ** 2)
    intercept = (sum_y - slope * sum_x) / n
    predicted = [slope * x + intercept for x in x_values]
    return slope, intercept, predicted

class Vector3D:
    def __init__(self, x, y, z):
        self.x = x
        self.y = y
        self.z = z

    def add(self, other):
        return Vector3D(self.x + other.x, self.y + other.y, self.z + other.z)

    def dot_product(self, other):
        return self.x * other.x + self.y * other.y + self.z * other.z

    def cross_product(self, other):
        return Vector3D(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x
        )

    def magnitude(self):
        return (self.x**2 + self.y**2 + self.z**2) ** 0.5

class ComplexNumber:
    def __init__(self, real, imag):
        self.real = real
        self.imag = imag

    def add(self, other):
        return ComplexNumber(self.real + other.real, self.imag + other.imag)

    def multiply(self, other):
        real_part = self.real * other.real - self.imag * other.imag
        imag_part = self.real * other.imag + self.imag * other.real
        return ComplexNumber(real_part, imag_part)

    def modulus(self):
        return (self.real**2 + self.imag**2) ** 0.5

    def conjugate(self):
        return ComplexNumber(self.real, -self.imag)

def moving_average(data, window_size):
    if window_size <= 0 or window_size > len(data):
        return []
    result = []
    current_sum = sum(data[:window_size])
    result.append(current_sum / window_size)
    for i in range(window_size, len(data)):
        current_sum = current_sum - data[i - window_size] + data[i]
        result.append(current_sum / window_size)
    return result

def z_score_normalization(data):
    n = len(data)
    if n < 2:
        return [0.0] * n
    mean = sum(data) / n
    variance = sum((x - mean) ** 2 for x in data) / n
    std_dev = variance ** 0.5
    if std_dev == 0:
        return [0.0] * n
    return [(x - mean) / std_dev for x in data]

def cosine_similarity(vec1, vec2):
    if len(vec1) != len(vec2):
        return 0.0
    dot_product = sum(a * b for a, b in zip(vec1, vec2))
    mag1 = sum(a ** 2 for a in vec1) ** 0.5
    mag2 = sum(b ** 2 for b in vec2) ** 0.5
    if mag1 == 0 or mag2 == 0:
        return 0.0
    return dot_product / (mag1 * mag2)

def trapezoidal_rule(func, a, b, n):
    h = (b - a) / n
    result = 0.5 * (func(a) + func(b))
    for i in range(1, n):
        result += func(a + i * h)
    result *= h
    return result

def simplified_base64_encode(text):
    b64_chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/"
    binary_string = "".join([bin(ord(c))[2:].zfill(8) for c in text])
    padding = (6 - len(binary_string) % 6) % 6
    binary_string += "0" * padding
    encoded = ""
    for i in range(0, len(binary_string), 6):
        chunk = binary_string[i:i+6]
        encoded += b64_chars[int(chunk, 2)]
    pad_chars = "=" * (padding // 2)
    return encoded + pad_chars

def url_encode(url_string):
    safe_chars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789-_.~"
    encoded = ""
    for char in url_string:
        if char in safe_chars:
            encoded += char
        elif char == ' ':
            encoded += '+'
        else:
            hex_val = hex(ord(char))[2:].upper()
            encoded += '%' + hex_val.zfill(2)
    return encoded

def word_wrap(text, max_length):
    words = text.split(' ')
    lines = []
    current_line = []
    current_length = 0
    for word in words:
        if current_length + len(word) + len(current_line) <= max_length:
            current_line.append(word)
            current_length += len(word)
        else:
            lines.append(" ".join(current_line))
            current_line = [word]
            current_length = len(word)
    if current_line:
        lines.append(" ".join(current_line))
    return "\n".join(lines)

def vigenere_cipher(text, key, encrypt=True):
    result = []

    text_int = [ord(i.upper()) - 65 for i in text if i.isalpha()]
    char_index = 0
    for char in text:
        if char.isalpha():
            shift = key_as_int[char_index % key_length]
            if not encrypt:
                shift = -shift

        else:
            result.append(char)
    return "".join(result)

def retry_decorator(max_retries, delay=1):
    def decorator(func):
        def wrapper(*args, **kwargs):
            import time
            attempts = 0
            while attempts < max_retries:
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    attempts += 1
                    if attempts >= max_retries:
                        raise e
                    time.sleep(delay)
        return wrapper
    return decorator

def check_sudoku_validity(board):
    def is_valid_group(group):
        nums = [n for n in group if n != '.']
        return len(nums) == len(set(nums))
    for row in board:
        if not is_valid_group(row):
            return False
    for col in range(9):
        if not is_valid_group([board[row][col] for row in range(9)]):
            return False
    for i in range(0, 9, 3):
        for j in range(0, 9, 3):
            square = [board[x][y] for x in range(i, i+3) for y in range(j, j+3)]
            if not is_valid_group(square):
                return False
    return True

def conways_game_of_life_step(grid):
    rows = len(grid)
    cols = len(grid[0])
    new_grid = [[0 for _ in range(cols)] for _ in range(rows)]
    for r in range(rows):
        for c in range(cols):
            live_neighbors = 0
            for i in range(-1, 2):
                for j in range(-1, 2):
                    if i == 0 and j == 0:
                        continue
                    nr, nc = r + i, c + j
                    if 0 <= nr < rows and 0 <= nc < cols:
                        live_neighbors += grid[nr][nc]
            if grid[r][c] == 1 and live_neighbors in [2, 3]:
                new_grid[r][c] = 1
            elif grid[r][c] == 0 and live_neighbors == 3:
                new_grid[r][c] = 1
    return new_grid

class DisjointSet:
    def __init__(self, size):
        self.parent = [i for i in range(size)]
        self.rank = [0] * size

    def find(self, i):
        if self.parent[i] == i:
            return i
        self.parent[i] = self.find(self.parent[i])
        return self.parent[i]

    def union(self, i, j):
        root_i = self.find(i)
        root_j = self.find(j)
        if root_i != root_j:
            if self.rank[root_i] < self.rank[root_j]:
                self.parent[root_i] = root_j
            elif self.rank[root_i] > self.rank[root_j]:
                self.parent[root_j] = root_i
            else:
                self.parent[root_j] = root_i
                self.rank[root_i] += 1

class CircularBuffer:
    def __init__(self, size):
        self.buffer = [None] * size
        self.size = size
        self.head = 0
        self.tail = 0
        self.is_full = False

    def enqueue(self, item):
        self.buffer[self.tail] = item
        if self.is_full:
            self.head = (self.head + 1) % self.size
        self.tail = (self.tail + 1) % self.size
        self.is_full = self.tail == self.head

    def dequeue(self):
        if not self.is_full and self.head == self.tail:
            return None
        item = self.buffer[self.head]
        self.buffer[self.head] = None
        self.head = (self.head + 1) % self.size
        self.is_full = False
        return item

class DoublyLinkedList:

        new_node = DoublyLinkedNode(new_data)
        new_node.next = self.head
        if self.head is not None:
            self.head.prev = new_node
        self.head = new_node

    def append(self, new_data):
        new_node = DoublyLinkedNode(new_data)
        if self.head is None:
            self.head = new_node
            return
        last = self.head
        while last.next:
            last = last.next
        last.next = new_node
        new_node.prev = last

def dictionary_to_mock_json(d, indent=4):
    def convert(obj, depth):
        spaces = " " * (depth * indent)
        if isinstance(obj, str):
            return f'"{obj}"'
        elif isinstance(obj, (int, float, bool)):
            return str(obj).lower() if isinstance(obj, bool) else str(obj)
        elif obj is None:
            return "null"
        elif isinstance(obj, list):
            items = [convert(item, depth + 1) for item in obj]
            inner = f",\n{spaces}    ".join(items)
            return f"[\n{spaces}    {inner}\n{spaces}]"
        elif isinstance(obj, dict):
            items = [f'"{k}": {convert(v, depth + 1)}' for k, v in obj.items()]
            inner = f",\n{spaces}    ".join(items)
            return f"{{\n{spaces}    {inner}\n{spaces}}}"
    return convert(d, 0)

            clusters[min_idx].append(point)
        new_centroids = []
        for cluster in clusters:
            if cluster:
                new_centroids.append(sum(cluster) / len(cluster))
            else:
                new_centroids.append(random.choice(data))
        centroids = new_centroids
    return clusters

def rabin_karp_search(text, pattern):
    n = len(text)
    m = len(pattern)
    if m == 0 or n < m:
        return []
    p_hash = hash(pattern)
    results = []
    for i in range(n - m + 1):
        window = text[i:i + m]
        if hash(window) == p_hash:
            if window == pattern:
                results.append(i)
    return results

function debounce(func, wait, immediate = false) {
    let timeout;
    return function() {
        const context = this;
        const args = arguments;
        const callNow = immediate && !timeout;
        clearTimeout(timeout);
        timeout = setTimeout(function() {
            timeout = null;
            if (!immediate) {
                func.apply(context, args);
            }
        }, wait);
        if (callNow) {
            func.apply(context, args);
        }
    };
}

function throttle(func, limit) {
    let inThrottle;
    return function() {
        const args = arguments;
        const context = this;
        if (!inThrottle) {
            func.apply(context, args);
            inThrottle = true;
            setTimeout(() => {
                inThrottle = false;
            }, limit);
        }
    };
}

function deepClone(obj) {
    if (obj === null || typeof obj !== 'object') {
        return obj;
    }
    if (obj instanceof Date) {
        return new Date(obj.getTime());
    }
    if (obj instanceof Array) {
        let copy = [];
        for (let i = 0; i < obj.length; i++) {
            copy[i] = deepClone(obj[i]);
        }
        return copy;
    }
    if (obj instanceof Object) {
        let copy = {};
        for (let attr in obj) {
            if (obj.hasOwnProperty(attr)) {
                copy[attr] = deepClone(obj[attr]);
            }
        }
        return copy;
    }
    throw new Error("Unable to copy obj!");
}

class EventEmitter {
    constructor() {
        this.events = {};
    }

    on(event, listener) {
        if (!this.events[event]) {
            this.events[event] = [];
        }
        this.events[event].push(listener);
    }

    emit(event, ...args) {
        if (this.events[event]) {
            this.events[event].forEach(listener => {
                listener.apply(this, args);
            });
        }
    }

    off(event, listenerToRemove) {
        if (!this.events[event]) return;
        this.events[event] = this.events[event].filter(
            listener => listener !== listenerToRemove
        );
    }
}

function heapSort(arr) {
    let n = arr.length;
    for (let i = Math.floor(n / 2) - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }
    for (let i = n - 1; i > 0; i--) {
        let temp = arr[0];
        arr[0] = arr[i];
        arr[i] = temp;
        heapify(arr, i, 0);
    }
    return arr;
    
    function heapify(arr, n, i) {
        let largest = i;
        let l = 2 * i + 1;
        let r = 2 * i + 2;
        if (l < n && arr[l] > arr[largest]) largest = l;
        if (r < n && arr[r] > arr[largest]) largest = r;
        if (largest !== i) {
            let swap = arr[i];
            arr[i] = arr[largest];
            arr[largest] = swap;
            heapify(arr, n, largest);
        }
    }
}

function shellSort(arr) {
    let n = arr.length;
    for (let gap = Math.floor(n / 2); gap > 0; gap = Math.floor(gap / 2)) {
        for (let i = gap; i < n; i += 1) {
            let temp = arr[i];
            let j;
            for (j = i; j >= gap && arr[j - gap] > temp; j -= gap) {
                arr[j] = arr[j - gap];
            }
            arr[j] = temp;
        }
    }
    return arr;
}

function binarySearchRecursive(arr, l, r, x) {
    if (r >= l) {
        let mid = l + Math.floor((r - l) / 2);
        if (arr[mid] === x) {
            return mid;
        }
        if (arr[mid] > x) {
            return binarySearchRecursive(arr, l, mid - 1, x);
        }
        return binarySearchRecursive(arr, mid + 1, r, x);
    }
    return -1;
}

function jumpSearch(arr, x) {
    let n = arr.length;
    let step = Math.floor(Math.sqrt(n));
    let prev = 0;
    while (arr[Math.min(step, n) - 1] < x) {
        prev = step;
        step += Math.floor(Math.sqrt(n));
        if (prev >= n) {
            return -1;
        }
    }
    while (arr[prev] < x) {
        prev++;
        if (prev === Math.min(step, n)) {
            return -1;
        }
    }
    if (arr[prev] === x) {
        return prev;
    }
    return -1;
}

function exponentialSearch(arr, x) {
    let n = arr.length;
    if (arr[0] === x) {
        return 0;
    }
    let i = 1;
    while (i < n && arr[i] <= x) {
        i = i * 2;
    }
    let l = Math.floor(i / 2);
    let r = Math.min(i, n - 1);
    while (l <= r) {
        let mid = l + Math.floor((r - l) / 2);
        if (arr[mid] === x) return mid;
        if (arr[mid] < x) l = mid + 1;
        else r = mid - 1;
    }
    return -1;
}

function matrixMultiplication(a, b) {
    let aNumRows = a.length;
    let aNumCols = a[0].length;
    let bNumCols = b[0].length;
    let m = new Array(aNumRows);
    for (let r = 0; r < aNumRows; ++r) {
        m[r] = new Array(bNumCols);
        for (let c = 0; c < bNumCols; ++c) {
            m[r][c] = 0;
            for (let i = 0; i < aNumCols; ++i) {
                m[r][c] += a[r][i] * b[i][c];
            }
        }
    }
    return m;
}

function sieveOfEratosthenes(max) {
    let sieve = new Array(max + 1).fill(true);
    sieve[0] = false;
    sieve[1] = false;
    for (let p = 2; p * p <= max; p++) {
        if (sieve[p] === true) {
            for (let i = p * p; i <= max; i += p) {
                sieve[i] = false;
            }
        }
    }
    let primes = [];
    for (let p = 2; p <= max; p++) {
        if (sieve[p]) {
            primes.push(p);
        }
    }
    return primes;
}

function gcdExtended(a, b) {
    if (a === 0) {
        return [b, 0, 1];
    }
    let result = gcdExtended(b % a, a);
    let gcd = result[0];
    let x1 = result[1];
    let y1 = result[2];
    let x = y1 - Math.floor(b / a) * x1;
    let y = x1;
    return [gcd, x, y];
}

function pascalTriangle(numRows) {
    let triangle = [];
    if (numRows === 0) {
        return triangle;
    }
    triangle.push([1]);
    for (let i = 1; i < numRows; i++) {
        let prevRow = triangle[i - 1];
        let newRow = [];
        newRow.push(1);
        for (let j = 1; j < prevRow.length; j++) {
            newRow.push(prevRow[j - 1] + prevRow[j]);
        }
        newRow.push(1);
        triangle.push(newRow);
    }
    return triangle;
}

function baseConverter(decNumber, base) {
    let remStack = [];
    let rem;
    let baseString = '';
    let digits = '0123456789ABCDEF';
    while (decNumber > 0) {
        rem = Math.floor(decNumber % base);
        remStack.push(rem);
        decNumber = Math.floor(decNumber / base);
    }
    while (remStack.length > 0) {
        baseString += digits[remStack.pop()];
    }
    return baseString || '0';
}

class Vector3D {
    constructor(x, y, z) {
        this.x = x;
        this.y = y;
        this.z = z;
    }

    add(other) {
        return new Vector3D(this.x + other.x, this.y + other.y, this.z + other.z);
    }

    dotProduct(other) {
        return this.x * other.x + this.y * other.y + this.z * other.z;
    }

    crossProduct(other) {
        return new Vector3D(
            this.y * other.z - this.z * other.y,
            this.z * other.x - this.x * other.z,
            this.x * other.y - this.y * other.x
        );
    }

    magnitude() {
        return Math.sqrt(this.x * this.x + this.y * this.y + this.z * this.z);
    }
}

class ComplexNumber {
    constructor(real, imag) {
        this.real = real;
        this.imag = imag;
    }

    add(other) {
        return new ComplexNumber(this.real + other.real, this.imag + other.imag);
    }

    multiply(other) {
        let r = this.real * other.real - this.imag * other.imag;
        let i = this.real * other.imag + this.imag * other.real;
        return new ComplexNumber(r, i);
    }

    modulus() {
        return Math.sqrt(this.real * this.real + this.imag * this.imag);
    }

    conjugate() {
        return new ComplexNumber(this.real, -this.imag);
    }
}

function movingAverage(data, windowSize) {
    let result = [];
    if (data.length < windowSize || windowSize <= 0) {
        return result;
    }
    let sum = 0;
    for (let i = 0; i < windowSize; i++) {
        sum += data[i];
    }
    result.push(sum / windowSize);
    for (let i = windowSize; i < data.length; i++) {
        sum = sum - data[i - windowSize] + data[i];
        result.push(sum / windowSize);
    }
    return result;
}

function zScoreNormalization(data) {
    let n = data.length;
    if (n < 2) return new Array(n).fill(0);
    let sum = data.reduce((a, b) => a + b, 0);
    let mean = sum / n;
    let varianceSum = 0;
    for (let i = 0; i < n; i++) {
        varianceSum += Math.pow(data[i] - mean, 2);
    }
    let stdDev = Math.sqrt(varianceSum / n);
    let result = [];
    for (let i = 0; i < n; i++) {
        if (stdDev === 0) result.push(0);
        else result.push((data[i] - mean) / stdDev);
    }
    return result;
}

function cosineSimilarity(vec1, vec2) {
    if (vec1.length !== vec2.length) return 0.0;
    let dotProduct = 0.0;
    let mag1 = 0.0;
    let mag2 = 0.0;
    for (let i = 0; i < vec1.length; i++) {
        dotProduct += vec1[i] * vec2[i];
        mag1 += vec1[i] * vec1[i];
        mag2 += vec2[i] * vec2[i];
    }
    mag1 = Math.sqrt(mag1);
    mag2 = Math.sqrt(mag2);
    if (mag1 === 0 || mag2 === 0) {
        return 0.0;
    }
    return dotProduct / (mag1 * mag2);
}

function compressString(str) {
    if (!str || str.length === 0) return "";
    let compressed = "";
    let count = 1;
    for (let i = 1; i < str.length; i++) {
        if (str[i] === str[i - 1]) {
            count++;
        } else {
            compressed += str[i - 1] + count;
            count = 1;
        }
    }
    compressed += str[str.length - 1] + count;
    if (compressed.length < str.length) {
        return compressed;
    }
    return str;
}

function caesarCipher(str, shift) {
    let result = '';
    for (let i = 0; i < str.length; i++) {
        let charCode = str.charCodeAt(i);
        if (charCode >= 65 && charCode <= 90) {
            result += String.fromCharCode(((charCode - 65 + shift) % 26) + 65);
        } else if (charCode >= 97 && charCode <= 122) {
            result += String.fromCharCode(((charCode - 97 + shift) % 26) + 97);
        } else {
            result += str.charAt(i);
        }
    }
    return result;
}

function vigenereCipher(text, key, encrypt = true) {
    let result = "";
    let j = 0;
    for (let i = 0; i < text.length; i++) {
        let c = text.charAt(i);
        if (c.match(/[a-z]/i)) {
            let code = text.charCodeAt(i);
            let isUpperCase = (code >= 65 && code <= 90);
            let shift = key.charCodeAt(j % key.length) - (key.charAt(j % key.length).toUpperCase() === key.charAt(j % key.length) ? 65 : 97);
            if (!encrypt) shift = -shift;
            let base = isUpperCase ? 65 : 97;
            let newCode = ((code - base + shift) % 26 + 26) % 26 + base;
            result += String.fromCharCode(newCode);
            j++;
        } else {
            result += c;
        }
    }
    return result;
}

function longestCommonPrefix(strs) {

            }
        }
    }

}

class LinkedListNode {
    constructor(val) {
        this.val = val;
        this.next = null;
    }
}

class LinkedList {
    constructor() {
        this.head = null;
    }

    add(val) {
        let newNode = new LinkedListNode(val);
        if (!this.head) {
            this.head = newNode;
            return;
        }
        let current = this.head;
        while (current.next) {
            current = current.next;
        }
        current.next = newNode;
    }

    reverse() {
        let prev = null;
        let current = this.head;
        let next = null;
        while (current !== null) {
            next = current.next;
            current.next = prev;
            prev = current;
            current = next;
        }
        this.head = prev;
    }
}

function detectCycleLinkedList(head) {
    if (!head || !head.next) {
        return false;
    }
    let slow = head;
    let fast = head;
    while (fast !== null && fast.next !== null) {
        slow = slow.next;
        fast = fast.next.next;
        if (slow === fast) {
            return true;
        }
    }
    return false;
}

function mergeSortedLinkedLists(l1, l2) {
    let dummy = new LinkedListNode(0);
    let current = dummy;
    while (l1 !== null && l2 !== null) {
        if (l1.val < l2.val) {
            current.next = l1;
            l1 = l1.next;
        } else {
            current.next = l2;
            l2 = l2.next;
        }
        current = current.next;
    }
    if (l1 !== null) {
        current.next = l1;
    } else if (l2 !== null) {
        current.next = l2;
    }
    return dummy.next;
}

class TrieNode {
    constructor() {
        this.children = {};
        this.isEndOfWord = false;
    }
}

class Trie {
    constructor() {
        this.root = new TrieNode();
    }

    insert(word) {
        let current = this.root;
        for (let i = 0; i < word.length; i++) {
            let ch = word[i];
            if (!current.children[ch]) {
                current.children[ch] = new TrieNode();
            }
            current = current.children[ch];
        }
        current.isEndOfWord = true;
    }

    search(word) {
        let current = this.root;
        for (let i = 0; i < word.length; i++) {
            let ch = word[i];
            if (!current.children[ch]) {
                return false;
            }
            current = current.children[ch];
        }
        return current.isEndOfWord;
    }
}

class LRUCache {
    constructor(capacity) {
        this.capacity = capacity;
        this.cache = new Map();
    }

    get(key) {
        if (!this.cache.has(key)) return -1;
        let val = this.cache.get(key);
        this.cache.delete(key);
        this.cache.set(key, val);
        return val;
    }

    put(key, value) {
        if (this.cache.has(key)) {
            this.cache.delete(key);
        }
        this.cache.set(key, value);
        if (this.cache.size > this.capacity) {
            let firstKey = this.cache.keys().next().value;
            this.cache.delete(firstKey);
        }
    }
}

class CircularBuffer {
    constructor(size) {
        this.buffer = new Array(size);
        this.size = size;
        this.head = 0;
        this.tail = 0;
        this.isFull = false;
    }

    enqueue(item) {
        this.buffer[this.tail] = item;
        if (this.isFull) {
            this.head = (this.head + 1) % this.size;
        }
        this.tail = (this.tail + 1) % this.size;
        this.isFull = this.head === this.tail;
    }

    dequeue() {
        if (!this.isFull && this.head === this.tail) {
            return null;
        }
        let item = this.buffer[this.head];
        this.buffer[this.head] = null;
        this.head = (this.head + 1) % this.size;
        this.isFull = false;
        return item;
    }
}

class DisjointSet {
    constructor(size) {
        this.parent = new Array(size);
        this.rank = new Array(size).fill(0);
        for (let i = 0; i < size; i++) {
            this.parent[i] = i;
        }
    }

    find(i) {
        if (this.parent[i] === i) {
            return i;
        }
        this.parent[i] = this.find(this.parent[i]);
        return this.parent[i];
    }

    union(i, j) {
        let rootI = this.find(i);
        let rootJ = this.find(j);
        if (rootI !== rootJ) {
            if (this.rank[rootI] < this.rank[rootJ]) {
                this.parent[rootI] = rootJ;
            } else if (this.rank[rootI] > this.rank[rootJ]) {
                this.parent[rootJ] = rootI;
            } else {
                this.parent[rootJ] = rootI;
                this.rank[rootI]++;
            }
        }
    }
}

class TicTacToe {

    }

            this.board[row][col] = this.currentPlayer;
            this.currentPlayer = this.currentPlayer === 'X' ? 'O' : 'X';
            return true;
        }
        return false;
    }

    checkWinner() {
        for (let i = 0; i < 3; i++) {
            if (this.board[i][0] === this.board[i][1] && this.board[i][1] === this.board[i][2] && this.board[i][0] !== ' ') return this.board[i][0];
            if (this.board[0][i] === this.board[1][i] && this.board[1][i] === this.board[2][i] && this.board[0][i] !== ' ') return this.board[0][i];
        }
        if (this.board[0][0] === this.board[1][1] && this.board[1][1] === this.board[2][2] && this.board[0][0] !== ' ') return this.board[0][0];
        if (this.board[0][2] === this.board[1][1] && this.board[1][1] === this.board[2][0] && this.board[0][2] !== ' ') return this.board[0][2];

    }

function conwaysGameOfLifeStep(grid) {
    let rows = grid.length;
    let cols = grid[0].length;
    let nextGrid = new Array(rows).fill(null).map(() => new Array(cols).fill(0));
    for (let r = 0; r < rows; r++) {
        for (let c = 0; c < cols; c++) {
            let liveNeighbors = 0;
            for (let i = -1; i <= 1; i++) {
                for (let j = -1; j <= 1; j++) {
                    if (i === 0 && j === 0) continue;
                    let nr = r + i;
                    let nc = c + j;
                    if (nr >= 0 && nr < rows && nc >= 0 && nc < cols) {
                        liveNeighbors += grid[nr][nc];
                    }
                }
            }
            if (grid[r][c] === 1 && (liveNeighbors === 2 || liveNeighbors === 3)) {
                nextGrid[r][c] = 1;
            } else if (grid[r][c] === 0 && liveNeighbors === 3) {
                nextGrid[r][c] = 1;
            }
        }
    }
    return nextGrid;
}

function passwordStrengthChecker(password) {
    let score = 0;
    if (!password) return score;
    if (password.length >= 8) score += 1;
    if (password.length >= 12) score += 1;
    let hasLower = /[a-z]/.test(password);
    let hasUpper = /[A-Z]/.test(password);
    let hasDigit = /[0-9]/.test(password);
    let hasSpecial = /[^a-zA-Z0-9]/.test(password);
    if (hasLower) score += 1;
    if (hasUpper) score += 1;
    if (hasDigit) score += 1;
    if (hasSpecial) score += 1;
    return score;
}

function isValidIPv4(ip) {
    if (typeof ip !== 'string') return false;
    let parts = ip.split('.');
    if (parts.length !== 4) return false;
    for (let i = 0; i < parts.length; i++) {
        let part = parts[i];
        if (!/^\d+$/.test(part)) return false;
        let num = parseInt(part, 10);
        if (num < 0 || num > 255) return false;
        if (part.length > 1 && part[0] === '0') return false;
    }
    return true;
}

function luhnValidator(cardNumber) {
    let value = cardNumber.replace(/\D/g, '');
    let sum = 0;
    let isEven = false;
    for (let n = value.length - 1; n >= 0; n--) {
        let digit = parseInt(value.charAt(n), 10);
        if (isEven) {
            digit *= 2;
            if (digit > 9) {
                digit -= 9;
            }
        }
        sum += digit;
        isEven = !isEven;
    }
    return (sum % 10) === 0;
}

function kMeans1D(data, k, maxIterations = 10) {
    if (data.length === 0 || k <= 0) return [];
    let centroids = [];
    for (let i = 0; i < k; i++) {
        centroids.push(data[Math.floor(Math.random() * data.length)]);
    }
    let clusters = [];
    for (let iter = 0; iter < maxIterations; iter++) {
        clusters = Array.from({length: k}, () => []);
        for (let i = 0; i < data.length; i++) {
            let point = data[i];
            let minDist = Infinity;
            let closestIdx = 0;
            for (let j = 0; j < k; j++) {
                let dist = Math.abs(point - centroids[j]);
                if (dist < minDist) {
                    minDist = dist;
                    closestIdx = j;
                }
            }
            clusters[closestIdx].push(point);
        }
        for (let j = 0; j < k; j++) {
            if (clusters[j].length > 0) {
                let sum = clusters[j].reduce((a, b) => a + b, 0);
                centroids[j] = sum / clusters[j].length;
            }
        }
    }
    return clusters;
}

class PIDController {
    constructor(kp, ki, kd) {
        this.kp = kp;
        this.ki = ki;
        this.kd = kd;
        this.prevError = 0;
        this.integral = 0;
    }

    compute(setpoint, measuredValue, dt) {
        let error = setpoint - measuredValue;
        this.integral += error * dt;
        let derivative = (error - this.prevError) / dt;
        let output = (this.kp * error) + (this.ki * this.integral) + (this.kd * derivative);
        this.prevError = error;
        return output;
    }
}

function advancedCurry(fn) {
  return function curried(...args) {
    if (args.length >= fn.length) {
      return fn.apply(this, args);
    } else {
      return function(...args2) {
        return curried.apply(this, args.concat(args2));
      };
    }
  };
}

function functionCompose(...fns) {
  if (fns.length === 0) {
    return arg => arg;
  }
  if (fns.length === 1) {
    return fns[0];
  }
  return fns.reduce((a, b) => (...args) => a(b(...args)));
}

function memoizeWithTTL(fn, ttl = 60000) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    const now = Date.now();
    if (cache.has(key)) {
      const { value, expiry } = cache.get(key);
      if (now < expiry) {
        return value;
      }
      cache.delete(key);
    }
    const result = fn.apply(this, args);
    cache.set(key, { value: result, expiry: now + ttl });
    return result;
  };
}

class AsyncMutex {
  constructor() {
    this.locked = false;
    this.queue = [];
  }
  lock() {
    return new Promise(resolve => {
      if (!this.locked) {
        this.locked = true;
        resolve();
      } else {
        this.queue.push(resolve);
      }
    });
  }
  unlock() {
    if (this.queue.length > 0) {
      const nextResolve = this.queue.shift();
      nextResolve();
    } else {
      this.locked = false;
    }
  }
}

class AsyncSemaphore {
  constructor(maxConcurrency) {
    this.tasks = [];
    this.counter = maxConcurrency;
  }
  acquire() {
    return new Promise(resolve => {
      if (this.counter > 0) {
        this.counter--;
        resolve();
      } else {
        this.tasks.push(resolve);
      }
    });
  }
  release() {
    if (this.tasks.length > 0) {
      const nextTask = this.tasks.shift();
      nextTask();
    } else {
      this.counter++;
    }
  }
}

async function promisePool(tasks, concurrency) {
  const results = [];
  const executing = [];
  for (const task of tasks) {
    const p = Promise.resolve().then(() => task());
    results.push(p);
    if (concurrency <= tasks.length) {
      const e = p.then(() => executing.splice(executing.indexOf(e), 1));
      executing.push(e);
      if (executing.length >= concurrency) {
        await Promise.race(executing);
      }
    }
  }
  return Promise.all(results);
}

function retryWithBackoff(fn, retries = 3, delay = 1000) {
  return new Promise((resolve, reject) => {
    const attempt = (n) => {
      fn().then(resolve).catch(err => {
        if (n === 0) {
          reject(err);
        } else {
          setTimeout(() => {
            attempt(n - 1);
          }, delay * (4 - n));
        }
      });
    };
    attempt(retries);
  });
}

function levenshteinDistance(a, b) {
  const matrix = [];
  for (let i = 0; i <= b.length; i++) {
    matrix[i] = [i];
  }
  for (let j = 0; j <= a.length; j++) {
    matrix[0][j] = j;
  }
  for (let i = 1; i <= b.length; i++) {
    for (let j = 1; j <= a.length; j++) {
      if (b.charAt(i - 1) === a.charAt(j - 1)) {
        matrix[i][j] = matrix[i - 1][j - 1];
      } else {
        matrix[i][j] = Math.min(
          matrix[i - 1][j - 1] + 1,
          matrix[i][j - 1] + 1,
          matrix[i - 1][j] + 1
        );
      }
    }
  }
  return matrix[b.length][a.length];
}

function kmpSearch(pattern, text) {
  if (pattern.length === 0) return 0;
  let lsp = new Array(pattern.length).fill(0);
  let j = 0;
  for (let i = 1; i < pattern.length; i++) {
    while (j > 0 && pattern[i] !== pattern[j]) {
      j = lsp[j - 1];
    }
    if (pattern[i] === pattern[j]) {
      j++;
    }
    lsp[i] = j;
  }
  let res = [];
  j = 0;
  for (let i = 0; i < text.length; i++) {
    while (j > 0 && text[i] !== pattern[j]) {
      j = lsp[j - 1];
    }
    if (text[i] === pattern[j]) {
      j++;
    }
    if (j === pattern.length) {
      res.push(i - j + 1);
      j = lsp[j - 1];
    }
  }
  return res;
}

function parseQueryString(queryString) {
  const query = {};
  if (!queryString) return query;
  const pairs = (queryString[0] === '?' ? queryString.substr(1) : queryString).split('&');
  for (let i = 0; i < pairs.length; i++) {
    const pair = pairs[i].split('=');
    const key = decodeURIComponent(pair[0]);
    const value = decodeURIComponent(pair[1] || '');
    if (query[key]) {
      if (Array.isArray(query[key])) {
        query[key].push(value);
      } else {
        query[key] = [query[key], value];
      }
    } else {
      query[key] = value;
    }
  }
  return query;
}

function buildQueryString(params) {
  const parts = [];
  for (const key in params) {
    if (params.hasOwnProperty(key)) {
      const val = params[key];
      if (Array.isArray(val)) {
        val.forEach(v => {
          parts.push(`${encodeURIComponent(key)}=${encodeURIComponent(v)}`);
        });
      } else if (val !== undefined && val !== null) {
        parts.push(`${encodeURIComponent(key)}=${encodeURIComponent(val)}`);
      }
    }
  }
  return parts.length > 0 ? '?' + parts.join('&') : '';
}

function parseCookieString(cookieStr) {

    if (parts.length >= 2) {

    }
  }
  return result;
}

class Graph {
  constructor() {
    this.adjacencyList = {};
  }
  addVertex(vertex) {
    if (!this.adjacencyList[vertex]) {
      this.adjacencyList[vertex] = [];
    }
  }
  addEdge(v1, v2) {
    if (this.adjacencyList[v1] && this.adjacencyList[v2]) {
      this.adjacencyList[v1].push(v2);
      this.adjacencyList[v2].push(v1);
    }
  }
  removeEdge(v1, v2) {
    this.adjacencyList[v1] = this.adjacencyList[v1].filter(v => v !== v2);
    this.adjacencyList[v2] = this.adjacencyList[v2].filter(v => v !== v1);
  }
  removeVertex(vertex) {
    while (this.adjacencyList[vertex].length) {
      const adjacentVertex = this.adjacencyList[vertex].pop();
      this.removeEdge(vertex, adjacentVertex);
    }
    delete this.adjacencyList[vertex];
  }
}

function depthFirstSearch(graph, start) {
  const result = [];
  const visited = {};
  const adjacencyList = graph.adjacencyList;
  (function dfs(vertex) {
    if (!vertex) return null;
    visited[vertex] = true;
    result.push(vertex);
    adjacencyList[vertex].forEach(neighbor => {
      if (!visited[neighbor]) {
        return dfs(neighbor);
      }
    });
  })(start);
  return result;
}

function breadthFirstSearch(graph, start) {
  const queue = [start];
  const result = [];
  const visited = {};
  let currentVertex;
  visited[start] = true;
  while (queue.length) {
    currentVertex = queue.shift();
    result.push(currentVertex);
    graph.adjacencyList[currentVertex].forEach(neighbor => {
      if (!visited[neighbor]) {
        visited[neighbor] = true;
        queue.push(neighbor);
      }
    });
  }
  return result;
}

class PriorityQueue {
  constructor() {
    this.values = [];
  }
  enqueue(val, priority) {
    this.values.push({ val, priority });
    this.sort();
  }
  dequeue() {
    return this.values.shift();
  }
  sort() {
    this.values.sort((a, b) => a.priority - b.priority);
  }
  isEmpty() {
    return this.values.length === 0;
  }
}

function dijkstraAlgorithm(graph, start, finish) {
  const nodes = new PriorityQueue();
  const distances = {};
  const previous = {};
  let path = [];
  let smallest;
  for (let vertex in graph.adjacencyList) {
    if (vertex === start) {
      distances[vertex] = 0;
      nodes.enqueue(vertex, 0);
    } else {
      distances[vertex] = Infinity;
      nodes.enqueue(vertex, Infinity);
    }
    previous[vertex] = null;
  }
  while (!nodes.isEmpty()) {
    smallest = nodes.dequeue().val;
    if (smallest === finish) {
      while (previous[smallest]) {
        path.push(smallest);
        smallest = previous[smallest];
      }
      break;
    }
    if (smallest || distances[smallest] !== Infinity) {
      for (let neighbor in graph.adjacencyList[smallest]) {
        let nextNode = graph.adjacencyList[smallest][neighbor];
        let candidate = distances[smallest] + nextNode.weight;
        let nextNeighbor = nextNode.node;
        if (candidate < distances[nextNeighbor]) {
          distances[nextNeighbor] = candidate;
          previous[nextNeighbor] = smallest;
          nodes.enqueue(nextNeighbor, candidate);
        }
      }
    }
  }
  return path.concat(start).reverse();
}

function topologicalSort(graph) {
  const visited = new Set();
  const stack = [];
  function explore(vertex) {
    visited.add(vertex);
    const neighbors = graph[vertex] || [];
    for (let i = 0; i < neighbors.length; i++) {
      if (!visited.has(neighbors[i])) {
        explore(neighbors[i]);
      }
    }
    stack.unshift(vertex);
  }
  for (let vertex in graph) {
    if (!visited.has(vertex)) {
      explore(vertex);
    }
  }
  return stack;
}

class BinarySearchTreeNode {
  constructor(value) {
    this.value = value;
    this.left = null;
    this.right = null;
  }
}

class BinarySearchTree {
  constructor() {
    this.root = null;
  }
  insert(value) {
    const newNode = new BinarySearchTreeNode(value);
    if (this.root === null) {
      this.root = newNode;
      return this;
    }
    let current = this.root;
    while (true) {
      if (value === current.value) return undefined;
      if (value < current.value) {
        if (current.left === null) {
          current.left = newNode;
          return this;
        }
        current = current.left;
      } else {
        if (current.right === null) {
          current.right = newNode;
          return this;
        }
        current = current.right;
      }
    }
  }
  find(value) {
    if (this.root === null) return false;
    let current = this.root, found = false;
    while (current && !found) {
      if (value < current.value) {
        current = current.left;
      } else if (value > current.value) {
        current = current.right;
      } else {
        found = true;
      }
    }
    if (!found) return undefined;
    return current;
  }
}

function grahamScanConvexHull(points) {
  if (points.length < 3) return points;
  let pivot = points[0];
  for (let i = 1; i < points.length; i++) {
    if (points[i].y < pivot.y || (points[i].y === pivot.y && points[i].x < pivot.x)) {
      pivot = points[i];
    }
  }
  points.sort((a, b) => {
    let angleA = Math.atan2(a.y - pivot.y, a.x - pivot.x);
    let angleB = Math.atan2(b.y - pivot.y, b.x - pivot.x);
    if (angleA === angleB) {
      let distA = Math.pow(a.x - pivot.x, 2) + Math.pow(a.y - pivot.y, 2);
      let distB = Math.pow(b.x - pivot.x, 2) + Math.pow(b.y - pivot.y, 2);
      return distA - distB;
    }
    return angleA - angleB;
  });
  let hull = [points[0], points[1]];
  for (let i = 2; i < points.length; i++) {
    while (hull.length > 1) {
      let top = hull.pop();
      let secondTop = hull[hull.length - 1];
      let crossProduct = (top.x - secondTop.x) * (points[i].y - secondTop.y) - (top.y - secondTop.y) * (points[i].x - secondTop.x);
      if (crossProduct > 0) {
        hull.push(top);
        break;
      }
    }
    hull.push(points[i]);
  }
  return hull;
}

function lineSegmentIntersection(p1, p2, p3, p4) {
  function ccw(A, B, C) {
    return (C.y - A.y) * (B.x - A.x) > (B.y - A.y) * (C.x - A.x);
  }
  let intersect = ccw(p1, p3, p4) !== ccw(p2, p3, p4) && ccw(p1, p2, p3) !== ccw(p1, p2, p4);
  if (!intersect) return null;
  let denominator = (p1.x - p2.x) * (p3.y - p4.y) - (p1.y - p2.y) * (p3.x - p4.x);
  if (denominator === 0) return null;
  let t = ((p1.x - p3.x) * (p3.y - p4.y) - (p1.y - p3.y) * (p3.x - p4.x)) / denominator;
  return {
    x: p1.x + t * (p2.x - p1.x),
    y: p1.y + t * (p2.y - p1.y)
  };
}

function pointInPolygon(point, vs) {
  let x = point[0], y = point[1];
  let inside = false;
  for (let i = 0, j = vs.length - 1; i < vs.length; j = i++) {
    let xi = vs[i][0], yi = vs[i][1];
    let xj = vs[j][0], yj = vs[j][1];
    let intersect = ((yi > y) !== (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi);
    if (intersect) inside = !inside;
  }
  return inside;
}

function matrixInverse2x2(matrix) {
  let det = matrix[0][0] * matrix[1][1] - matrix[0][1] * matrix[1][0];
  if (det === 0) {
    return null;
  }
  let invDet = 1 / det;
  return [
    [matrix[1][1] * invDet, -matrix[0][1] * invDet],
    [-matrix[1][0] * invDet, matrix[0][0] * invDet]
  ];
}

function haversineDistance(lat1, lon1, lat2, lon2) {
  const R = 6371; 
  const dLat = (lat2 - lat1) * Math.PI / 180;
  const dLon = (lon2 - lon1) * Math.PI / 180;
  const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
            Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
            Math.sin(dLon / 2) * Math.sin(dLon / 2);
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  const d = R * c;
  return d;
}

class SingleNeuronPerceptron {
  constructor(learningRate = 0.1, iterations = 100) {
    this.learningRate = learningRate;
    this.iterations = iterations;
    this.weights = [];
    this.bias = 0;
  }
  train(inputs, labels) {
    this.weights = new Array(inputs[0].length).fill(0);
    for (let i = 0; i < this.iterations; i++) {
      for (let j = 0; j < inputs.length; j++) {
        let prediction = this.predict(inputs[j]);
        let error = labels[j] - prediction;
        for (let k = 0; k < this.weights.length; k++) {
          this.weights[k] += this.learningRate * error * inputs[j][k];
        }
        this.bias += this.learningRate * error;
      }
    }
  }
  predict(input) {
    let sum = this.bias;
    for (let i = 0; i < this.weights.length; i++) {
      sum += input[i] * this.weights[i];
    }
    return sum >= 0 ? 1 : 0;
  }
}

function tfIdfCalculator(documents) {
  let termFrequency = [];
  let documentFrequency = {};
  documents.forEach(doc => {
    let words = doc.toLowerCase().match(/\b\w+\b/g) || [];
    let tf = {};
    let uniqueWords = new Set(words);
    words.forEach(word => {
      tf[word] = (tf[word] || 0) + 1 / words.length;
    });
    termFrequency.push(tf);
    uniqueWords.forEach(word => {
      documentFrequency[word] = (documentFrequency[word] || 0) + 1;
    });
  });
  let tfIdf = [];
  let numDocs = documents.length;
  termFrequency.forEach(tf => {
    let docTfIdf = {};
    for (let word in tf) {
      let idf = Math.log(numDocs / (1 + documentFrequency[word]));
      docTfIdf[word] = tf[word] * idf;
    }
    tfIdf.push(docTfIdf);
  });
  return tfIdf;
}

class LocalStorageWithExpiry {
  static setItem(key, value, expiryInMinutes) {
    const now = new Date();
    const item = {
      value: value,
      expiry: now.getTime() + expiryInMinutes * 60000,
    };
    localStorage.setItem(key, JSON.stringify(item));
  }
  static getItem(key) {
    const itemStr = localStorage.getItem(key);
    if (!itemStr) return null;
    const item = JSON.parse(itemStr);
    const now = new Date();
    if (now.getTime() > item.expiry) {
      localStorage.removeItem(key);
      return null;
    }
    return item.value;
  }
  static removeItem(key) {
    localStorage.removeItem(key);
  }
}

function deepMergeObjects(target, source) {
  let output = Object.assign({}, target);
  if (isObject(target) && isObject(source)) {
    Object.keys(source).forEach(key => {
      if (isObject(source[key])) {
        if (!(key in target)) {
          Object.assign(output, { [key]: source[key] });
        } else {
          output[key] = deepMergeObjects(target[key], source[key]);
        }
      } else {
        Object.assign(output, { [key]: source[key] });
      }
    });
  }
  return output;
  function isObject(item) {
    return (item && typeof item === 'object' && !Array.isArray(item));
  }
}

function flattenObject(obj, prefix = '') {
  let flattened = {};
  for (let key in obj) {
    if (obj.hasOwnProperty(key)) {
      let newKey = prefix ? prefix + '.' + key : key;
      if (typeof obj[key] === 'object' && obj[key] !== null && !Array.isArray(obj[key])) {
        Object.assign(flattened, flattenObject(obj[key], newKey));
      } else {
        flattened[newKey] = obj[key];
      }
    }
  }
  return flattened;
}

function unflattenObject(data) {
  if (Object(data) !== data || Array.isArray(data)) return data;
  let result = {};
  for (let p in data) {
    let keys = p.split('.');
    let last = keys.pop();
    let current = result;
    for (let key of keys) {
      current[key] = current[key] || {};
      current = current[key];
    }
    current[last] = data[p];
  }
  return result;
}

function generateUUIDv4() {
  return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
    let r = Math.random() * 16 | 0,
        v = c === 'x' ? r : (r & 0x3 | 0x8);
    return v.toString(16);
  });
}

function parseCSVString(csvText, delimiter = ',') {
  let lines = csvText.split('\n');
  let result = [];
  let headers = lines[0].split(delimiter).map(header => header.trim());
  for (let i = 1; i < lines.length; i++) {
    if (!lines[i].trim()) continue;
    let obj = {};
    let currentline = lines[i].split(delimiter);
    for (let j = 0; j < headers.length; j++) {
      obj[headers[j]] = currentline[j] ? currentline[j].trim() : '';
    }
    result.push(obj);
  }
  return result;
}

function stringifyToCSV(objArray, delimiter = ',') {
  if (!objArray || !objArray.length) return '';
  let array = typeof objArray !== 'object' ? JSON.parse(objArray) : objArray;
  let str = '';
  let keys = Object.keys(array[0]);
  str += keys.join(delimiter) + '\r\n';
  for (let i = 0; i < array.length; i++) {
    let line = '';
    for (let index in keys) {
      if (line !== '') line += delimiter;
      let value = array[i][keys[index]];
      if (typeof value === 'string' && value.includes(delimiter)) {
        value = `"${value}"`;
      }
      line += value;
    }
    str += line + '\r\n';
  }
  return str;
}

function formatBytesReadable(bytes, decimals = 2) {
  if (bytes === 0) return '0 Bytes';
  const k = 1024;
  const dm = decimals < 0 ? 0 : decimals;
  const sizes = ['Bytes', 'KB', 'MB', 'GB', 'TB', 'PB', 'EB', 'ZB', 'YB'];
  const i = Math.floor(Math.log(bytes) / Math.log(k));
  return parseFloat((bytes / Math.pow(k, i)).toFixed(dm)) + ' ' + sizes[i];
}

function shuffleArrayFisherYates(array) {
  let currentIndex = array.length, temporaryValue, randomIndex;
  while (0 !== currentIndex) {
    randomIndex = Math.floor(Math.random() * currentIndex);
    currentIndex -= 1;
    temporaryValue = array[currentIndex];
    array[currentIndex] = array[randomIndex];
    array[randomIndex] = temporaryValue;
  }
  return array;
}

function chunkArrayItems(array, size) {

  for (let i = 0; i < array.length; i += size) {
    result.push(array.slice(i, i + size));
  }
  return result;
}

function isValidEmailAddress(email) {
  if (typeof email !== 'string') return false;
  let re = /^(([^<>()\[\]\\.,;:\s@"]+(\.[^<>()\[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
  return re.test(email.toLowerCase());
}

function toCamelCaseString(str) {
  return str.replace(/(?:^\w|[A-Z]|\b\w)/g, function(word, index) {
    return index === 0 ? word.toLowerCase() : word.toUpperCase();
  }).replace(/\s+/g, '').replace(/[^a-zA-Z0-9]/g, '');
}

function toSnakeCaseString(str) {
  return str.match(/[A-Z]{2,}(?=[A-Z][a-z]+[0-9]*|\b)|[A-Z]?[a-z]+[0-9]*|[A-Z]|[0-9]+/g)
    .map(x => x.toLowerCase())
    .join('_');
}

function toKebabCaseString(str) {
  return str.match(/[A-Z]{2,}(?=[A-Z][a-z]+[0-9]*|\b)|[A-Z]?[a-z]+[0-9]*|[A-Z]|[0-9]+/g)
    .map(x => x.toLowerCase())
    .join('-');
}

function getArrayIntersection(arr1, arr2) {
  let set1 = new Set(arr1);
  let set2 = new Set(arr2);
  let intersection = new Set([...set1].filter(x => set2.has(x)));
  return Array.from(intersection);
}

function getArrayDifference(arr1, arr2) {
  let set1 = new Set(arr1);
  let set2 = new Set(arr2);
  let difference = new Set([...set1].filter(x => !set2.has(x)));
  return Array.from(difference);
}

function getArraySymmetricDifference(arr1, arr2) {
  let diff1 = getArrayDifference(arr1, arr2);
  let diff2 = getArrayDifference(arr2, arr1);
  return diff1.concat(diff2);
}

function clampValueInRange(value, min, max) {
  if (value === undefined || isNaN(value)) return 0;
  let lower = Math.min(min, max);
  let upper = Math.max(min, max);
  return Math.min(Math.max(value, lower), upper);
}
